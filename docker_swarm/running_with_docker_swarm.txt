# Init or join a swarm
docker swarm init

# Create registry service
docker service create --name exampleup-registry --mode global --publish 5000:5000 markwatsonatx/exampleup-registry:latest

# Create peristent volume
docker volume create --name example1-volume

# Run exampleup
docker-compose -f docker-compose-example1.yml -p example1 up

# Access log and editor with assigned port
http://localhost:DYNAMIC_PORT

# docker-compose container URLS
http://port-exposed-through-dockercompose.localhost:DYNAMIC_PORT
http://33000.localhost:32830

# Stop
docker-compose -f docker-compose-example1.yml -p example1 down
